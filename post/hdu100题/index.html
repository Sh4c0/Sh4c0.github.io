        ans1 = 0, ans2 = 0;
        if (l > r) {
            swap(l, r);

const int inf = 0x3f3f3f3f;

int main() {
    int n, m, mmax, mmin, t;
    double ans;
    while (~scanf("%d", &n)) {
        m = n;
        mmax = 0;
        mmin = inf;
        ans = 0;
        while (m--) {
            scanf("%d", &t);
            mmax = max(mmax, t);
            mmin = min(mmin, t);
            ans += t;
        }
        printf("%.2f\n", (ans - mmax - mmin) / (n - 2));
    }
    return 0;
}

hdu2028

传送门

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

inline ll gcd(ll a, ll b) {
    ll t;
    while (b != 0) {
        t = a;
        a = b;
        b = t % b;
    }
    return a;
}

inline ll lcm(ll a, ll b) {
    return a * b / gcd(a, b);
}

int main() {
    ll n, m, ans;
    while (~scanf("%lld", &n)) {
        scanf("%lld", &ans);
        n--;
        while (n--) {
            scanf("%lld", &m);
            ans = lcm(ans, m);
        }
        printf("%lld\n", ans);
    }
    return 0;
}
